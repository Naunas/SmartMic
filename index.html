# Meet Smart Mic AI: Your Real-Time Audio Companion

mic = SmartMic(
    privacy_mode="local",
    integration=["Zoom", "Teams", "YouTube", "Podcasts"]
)

# Start listening for insights
mic.listen()

# Instantly transcribe and summarize conversations
@mic.on_transcribe
def handle(text):
    if mic.is_question(text):
        answer = mic.answer(text)
        mic.display(answer)
    elif mic.is_action(text):
        mic.perform_action(text)
    else:
        mic.summarize(text)
